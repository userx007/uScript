# =============================================================================
# test_eval.script
#
# Comprehensive feature test for the EVAL expression evaluator.
#
# Coverage:
#   1.  Integration points  — ?=  /  IF GOTO  /  REPEAT UNTIL
#   2.  Type inference      — STR / NUM / VER / BOOL (auto-detected)
#   3.  Explicit type hints — inline  op:TYPE  /  postfix  :TYPE  /  : TYPE
#   4.  String comparisons  — EQ / eq / NE / ne / == / != (case-sensitive)
#   5.  Numeric comparisons — == != < <= > >=
#   6.  Version comparisons — == != < <= > >=
#   7.  Boolean comparisons — == != (TRUE/FALSE/!TRUE/!FALSE, case-insensitive)
#   8.  Lone boolean atoms  — TRUE / FALSE / !TRUE / !FALSE
#   9.  Compound expressions — && / ||  (precedence: && binds tighter than ||)
#   10. Short-circuit evaluation
#   11. Macro expansion inside EVAL
#   12. REPEAT UNTIL EVAL — loop termination
#   13. Negative / error cases — wrong type, missing operand, bad syntax
#
# Convention:
#   ok_<N>   ?=  EVAL ...        expected TRUE
#   fail_<N> ?=  EVAL ...        expected FALSE
#   PRINT $ok_<N>   should print TRUE
#   PRINT $fail_<N> should print FALSE
#
# No plugins are needed — EVAL, PRINT, REPEAT, IF/GOTO/LABEL, FORMAT
# are all native interpreter statements.
# =============================================================================

# 20. END-TO-END FLOW: condition drives branching, result stored in macro

PRINT --- 20. END-TO-END FLOW ---

# Is current firmware below threshold?

# Is current firmware below target?

# Compound: needs update AND below threshold

# Branch on it

PRINT ok_20_already_current

LABEL do_update
PRINT ok_20_update_needed
